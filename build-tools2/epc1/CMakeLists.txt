"zeromq 4.2.1-1"
"zita-alsa-pcmi 0.2.0-3"
"zita-resampler 1.3.0-5"
"zlib 1:1.2.11-1"
"zsh 5.3.1-2"
"zvbi 0.2.35-1"
)

foreach(PACKAGE ${EPC1_PACKAGES})
    separate_arguments(PACKAGE)
    list(GET PACKAGE 0 PACKAGE_NAME)
    list(GET PACKAGE 1 PACKAGE_VERSION)
    SET(FILE "${PACKAGE_NAME}-${PACKAGE_VERSION}-x86_64.pkg.tar.xz")
    SET(TARGET "${CMAKE_CURRENT_BINARY_DIR}/${FILE}")
    SET(NL_URL "${DOWNLOAD_LOCATION}/epc1/${FILE}")
    SET(URL "https://archive.archlinux.org/packages/.all/${FILE}")
    string(SHA512 SHA ${FILE})
    
    add_custom_command(
        OUTPUT "${SHA}"
        COMMAND wget ${NL_URL} -O "${SHA}" || wget ${URL} -O "${SHA}"
        COMMAND ln -s "${SHA}" "${FILE}"
    )

    list(APPEND EPC1_DOWNLOADED_PACKAGES "${SHA}")
endforeach()

add_custom_target(epc1-download-packages DEPENDS ${EPC1_DOWNLOADED_PACKAGES})
